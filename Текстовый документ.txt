Задание 5. Сконфигурируйте ansible на сервере BR-SRV:

BR-SRV
apt install ansible sshpass -y
mkdir -p /etc/ansible
nano /etc/ansible/ansible.cfg
[defaults]
host_key_checking=False
nano /etc/ansible/hosts
[HQ]
192.168.100.1  ansible_user=net_admin  ansible_password=P@ssword
192.168.100.2  ansible_user=sshuser    ansible_password=P@ssword    ansible_port=2026
192.168.100.35 ansible_user=sshuser    ansible_password=P@ssword
[br]
192.168.200.1 ansible_user=net_admin  ansible_password=P@ssword
[all:vars]
ansible_python_interpreter=/usr/bi/python3.13
